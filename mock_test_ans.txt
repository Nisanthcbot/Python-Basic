"""
Q6 (7.5 Marks)

ATM Withdrawal
If withdrawal <= 0 → Invalid
If withdrawal > balance → Insufficient
Else deduct and print remaining balance

"""

withdrawal = int(input('Enter the Withdrawal Amount :'))
balance = 10000


if withdrawal <0:
    print("Invalid ")

elif withdrawal > balance:
    print("Insufficient")


else:
    remaining = balance - withdrawal
    print(f"Withdrawal Amount is {withdrawal} and the remaining Amount is {remaining}")
